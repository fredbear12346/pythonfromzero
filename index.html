üéÆ Project: Number Guessing Game

25 min

Let's build a complete game using everything you've learned!

Complete Game Code

import random

def play_game():
    print("=" * 40)
    print("üéØ NUMBER GUESSING GAME")
    print("=" * 40)
    
    # Generate secret number
    secret = random.randint(1, 100)
    attempts = 0
    max_attempts = 7
    
    print(f"\nI'm thinking of a number between 1 and 100")
    print(f"You have {max_attempts} attempts. Good luck!\n")
    
    while attempts < max_attempts:
        try:
            guess = int(input("Your guess: "))
            attempts += 1
            
            if guess == secret:
                print(f"\nüéâ CORRECT! You got it in {attempts} attempts!")
                return True
            elif guess < secret:
                print("üìà Too LOW! Guess higher.")
            else:
                print("üìâ Too HIGH! Guess lower.")
            
            remaining = max_attempts - attempts
            if remaining > 0:
                print(f"   ({remaining} attempts left)\n")
                
        except ValueError:
            print("Please enter a valid number!")
    
    print(f"\nüò¢ Game Over! The number was {secret}")
    return False

# Play the game
while True:
    play_game()
    again = input("\nPlay again? (yes/no): ")
    if again.lower() != "yes":
        print("Thanks for playing! üëã")
        break

class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner
        self.balance = balance
    
    def deposit(self, amount):
        self.balance += amount
        print(f"Deposited ${amount}. Balance: ${self.balance}")
    
    def withdraw(self, amount):
        if amount <= self.balance:
            self.balance -= amount
            print(f"Withdrew ${amount}. Balance: ${self.balance}")
        else:
            print("Insufficient funds!")

# Use it

class Car(Vehicle):

Practical Example: Timer

import time

def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} took {end-start:.4f} seconds")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(1)
    print("Done!")

slow_function()  # Done! + time info

üéÆ Project: Rock Paper Scissors

30 min

Build a complete game with everything you've learned!

import random

def get_choice():
    choices = {"r": "rock", "p": "paper", "s": "scissors"}
    while True:
        user = input("Rock (r), Paper (p), Scissors (s): ").lower()
        if user in choices:
            return choices[user]
        print("Invalid choice!")

def determine_winner(user, computer):
    if user == computer:
        return "tie"
    wins = {"rock": "scissors", "paper": "rock", "scissors": "paper"}
    return "user" if wins[user] == computer else "computer"

def play():
    scores = {"user": 0, "computer": 0}
    
    print("üéÆ ROCK PAPER SCISSORS")
    print("First to 3 wins!\n")
    
    while max(scores.values()) < 3:
        user = get_choice()
        computer = random.choice(["rock", "paper", "scissors"])
        
        print(f"You: {user} | Computer: {computer}")
        
        winner = determine_winner(user, computer)
        if winner == "tie":
            print("Tie!\n")
        else:
            scores[winner] += 1
            print(f"{winner.title()} wins! Score: {scores}\n")
    
    final = "You win! üéâ" if scores["user"] > scores["computer"] else "Computer wins!"
    print(final)

play()

üì± Contact Manager App

import json
from pathlib import Path
from datetime import datetime

class ContactManager:
    def __init__(self, filename="contacts.json"):
        self.file = Path(filename)
        self.contacts = self.load()
    
    def load(self):
        if self.file.exists():
            return json.loads(self.file.read_text())
        return {}
    
    def save(self):
        self.file.write_text(json.dumps(self.contacts, indent=2))
    
    def add(self, name, phone, email=""):
        self.contacts[name] = {
            "phone": phone,
            "email": email,
            "created": datetime.now().isoformat()
        }
        self.save()
        print(f"‚úÖ Added {name}")
    
    def search(self, query):
        results = [n for n in self.contacts if query.lower() in n.lower()]
        return results
    
    def delete(self, name):
        if name in self.contacts:
            del self.contacts[name]
            self.save()
            print(f"‚ùå Deleted {name}")
    
    def list_all(self):
        for name, info in self.contacts.items():
            print(f"\nüìû {name}")
            print(f"   Phone: {info['phone']}")
            print(f"   Email: {info.get('email', 'N/A')}")

def main():
    cm = ContactManager()
    
    while True:
        print("\n=== CONTACT MANAGER ===")
        print("1. Add contact")
        print("2. Search contacts")
        print("3. List all contacts")
        print("4. Delete contact")
        print("5. Exit")
        
        choice = input("\nChoice: ")
        
        if choice == "1":
            name = input("Name: ")
            phone = input("Phone: ")
            email = input("Email (optional): ")
            cm.add(name, phone, email)
        elif choice == "2":
            query = input("Search: ")
            results = cm.search(query)
            print(f"Found: {results}")
        elif choice == "3":
            cm.list_all()
        elif choice == "4":
            name = input("Name to delete: ")
            cm.delete(name)
        elif choice == "5":
            print("Goodbye! üëã")
            break

if __name__ == "__main__":
    main()
